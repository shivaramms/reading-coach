Set up Google Sheets logging

This uses a Google Apps Script — a tiny piece of code that runs inside Google and writes to your Sheet. No extra accounts needed, just a Google account.

Step 1 — Create the Sheet

Go to sheets.new to create a new Google Sheet. Name it "Reading Coach".

Step 2 — Open Apps Script

In your Sheet, click Extensions → Apps Script. Delete any code in the editor and paste this:

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet()
                  .getActiveSheet();

    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        'Name','Date','Book','Author',
        'Overall','Comprehension','Language','Structure','Mechanics',
        'Summary','Top Tips','Weekly Challenge','Encouragement',
        'Comp Feedback','Lang Feedback','Struct Feedback','Mech Feedback','Spelling'
      ]);
      sheet.getRange(1,1,1,18).setFontWeight('bold').setBackground('#E8F5E9');
    }

    var name = data.name || 'Unknown';
    sheet.appendRow([
      name, data.date, data.title, data.author,
      data.overall, data.comprehension, data.language,
      data.structure, data.mechanics, data.summary,
      (data.top_tips || []).join(' | '),
      data.weekly_challenge, data.encouragement,
      data.comprehension_feedback, data.language_feedback,
      data.structure_feedback, data.mechanics_feedback,
      data.spelling_corrections || '[]'
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error',msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    var action = e.parameter.action || 'get';
    var name = e.parameter.name || '';
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

    if (action === 'delete' && name) {
      var rows = sheet.getDataRange().getValues();
      for (var i = rows.length - 1; i >= 1; i--) {
        if (rows[i][0] && rows[i][0].toString().toLowerCase() === name.toLowerCase()) {
          sheet.deleteRow(i + 1);
        }
      }
      return ContentService
        .createTextOutput(JSON.stringify({status:'deleted'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    var rows = sheet.getDataRange().getValues();
    if (rows.length <= 1) {
      return ContentService
        .createTextOutput(JSON.stringify([]))
        .setMimeType(ContentService.MimeType.JSON);
    }
    var results = [];
    for (var i = 1; i < rows.length; i++) {
      var r = rows[i];
      var rowName = r[0] ? r[0].toString() : '';
      if (name && rowName.toLowerCase() !== name.toLowerCase()) continue;
      results.push({
        name:r[0], date:r[1], title:r[2], author:r[3],
        overall:r[4], comprehension:r[5], language:r[6],
        structure:r[7], mechanics:r[8], summary:r[9],
        top_tips:r[10]?r[10].toString().split(' | '):[],
        weekly_challenge:r[11], encouragement:r[12],
        comprehension_feedback:r[13], language_feedback:r[14],
        structure_feedback:r[15], mechanics_feedback:r[16],
        spelling_corrections:r[17]?r[17].toString():'[]'
      });
    }
    return ContentService
      .createTextOutput(JSON.stringify(results))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

Step 3 — Deploy it

Click Deploy → New deployment. Set:

  • Type: Web app
  • Execute as: Me
  • Who has access: Anyone

Click Deploy, approve permissions, then copy the Web app URL.

Step 4 — Paste the URL

Paste that URL into the "Google Sheets URL" field in Reading Coach settings and save. Every evaluation will now appear as a new row in your Sheet automatically.
